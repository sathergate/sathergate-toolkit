notifykit

Unified notifications for Next.js

SMS, email, push — one API. Swap providers, not code.

$ npm install notifykit

Quick Start

Get started in three steps

Configure your providers, send notifications, and add templates — all with a single, unified API.

1

Configure providers

lib/herald.ts 
import { createHerald } from "notifykit";
import { createTwilioProvider } from "notifykit/adapters/twilio";
import { createResendProvider } from "notifykit/adapters/resend";

export const herald = createHerald({
  providers: [
    createTwilioProvider({ accountSid: "...", authToken: "...", from: "+1..." }),
    createResendProvider({ apiKey: "re_...", from: "noreply@example.com" }),
  ],
});
2

Send notifications

app/api/notify/route.ts 
await herald.send({
  to: "+15559876543",
  channel: "sms",
  body: "Your order shipped!",
});

await herald.send({
  to: "user@example.com",
  channel: "email",
  subject: "Welcome",
  body: "Thanks for signing up.",
});
3

Use templates

lib/herald.ts 
const herald = createHerald({
  providers: [twilio, resend],
  templates: {
    "order-shipped": {
      channel: "sms",
      body: "Hi {{name}}, order #{{orderId}} shipped!",
    },
  },
});

await herald.notify("order-shipped", {
  to: "+1...", data: { name: "Alice", orderId: "123" },
});
